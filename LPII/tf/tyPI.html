Start ::= cmd <EOL>
cmd ::= state ( <SEQ> cmd )*
state ::= callFun ( <ASSIGN> decl | <LPAR> ( decl ( <COMMA> decl )* )? <RPAR> )
| <WHILE> <LPAR> decl <RPAR> <DO> cmd <END>
| <IF> decl <THEN> cmd ( <ELSE> cmd )?
| <DECL> ( <ID> <EQ> decl )* <IN> cmd <END>
| <PRINT> <LPAR> decl <RPAR>
| <PRINTLN> <LPAR> <RPAR>
| <FOREACH> <ID> <IN> exp <DO> sequence <END>
decl ::= logicalOr
| <DECL> declExp
declExp ::= ( ( <ID> <EQ> exp )* <IN> exp <END> )
logicalOr ::= logicalAnd ( <OR> logical )*
logicalAnd ::= comp ( <AND> comp )*
| <IF> logical <THEN> comp ( <ELSE> comp )? <END>
comp ::= exp ( <EQ> exp | <GREATER> exp | <SMALLER> exp | <GREATEROREQUAL> exp | <SMALLEROREQUAL> exp | <DIFF> exp )*
exp ::= term ( <PLUS> term | <MINUS> term )*
term ::= callFun ( <MULTIPLY> factor | <DIVISION> factor | <MOD> factor )*
callFun ::= factor ( ( <LPAR> ( factor ( <COMMA> factor )* )? <RPAR> ) | <DOT> <ID> | <LBRACK> factor <RBRACK> )*
factor ::= <CONSTANT>
| <LPAR> logical <RPAR>
| <ID>
| <NOT> factor
| <VAR> <LPAR> logical <RPAR>
| <DEREF> factor
| <TRUE>
| <FALSE>
| <READINT> <LPAR> <RPAR>
| <READSTRING> <LPAR> <RPAR>
| <STRING>
| <DO> cmd <RETURN> logical <END>
| <MINUS> factor
| <LBRACK> <ID> <EQ> logical ( <COMMA> <ID> <EQ> logical )* <RBRACK>
| <FUN> fun
| <PROC> proc
| <ARRAY> <LPAR> exp <COMMA> exp <RPAR>
| <LENGTH> <LPAR> exp <RPAR>
fun ::= ( <ID> ( <COMMA> <ID> )* )? <IS> logical <END>
proc ::= ( <ID> ( <COMMA> <ID> )* )? <IS> state <END>
